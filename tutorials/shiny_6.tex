		\vspace{1em}

		\vspace{1em}

		\begin{exampleblock}{}
		\begin{BVerbatim}
library(shiny)
runGist("https://gist.github.com/MyName/identifier")
		\end{BVerbatim}
		\end{exampleblock}{}
